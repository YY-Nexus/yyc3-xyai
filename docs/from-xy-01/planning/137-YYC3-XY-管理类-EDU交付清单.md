```sql
    USING (institution_id = current_setting('app.current_institution', true));

-- 敏感数据加密
CREATE TABLE encrypted_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    sensitive_data VARBINARY(255) NOT NULL,
    iv VARBINARY(16) NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- 创建加密函数
DELIMITER //
CREATE FUNCTION encrypt_data(data VARCHAR(255), key VARCHAR(255)) 
RETURNS VARBINARY(255)
DETERMINISTIC
BEGIN
    SET @iv = UNHEX(SHA2(UUID(), 256));
    SET @encrypted = AES_ENCRYPT(data, key, @iv);
    RETURN CONCAT(@iv, @encrypted);
END //
DELIMITER ;

-- 创建解密函数
DELIMITER //
CREATE FUNCTION decrypt_data(encrypted_data VARBINARY(255), key VARCHAR(255)) 
RETURNS VARCHAR(255)
DETERMINISTIC
BEGIN
    SET @iv = SUBSTRING(encrypted_data, 1, 16);
    SET @ciphertext = SUBSTRING(encrypted_data, 17);
    RETURN AES_DECRYPT(@ciphertext, key, @iv);
END //
DELIMITER ;
```

### 6.5 数据库备份与恢复策略

```sql
-- 创建备份表空间
CREATE TABLESPACE backup LOCATION '/var/lib/mysql/backups';

-- 创建存储过程进行备份
DELIMITER //
CREATE PROCEDURE backup_database()
BEGIN
    DECLARE backup_file VARCHAR(100);
    DECLARE backup_timestamp VARCHAR(20);
    
    SET backup_timestamp = DATE_FORMAT(NOW(), '%Y%m%d_%H%i%s');
    SET backup_file = CONCAT('education_backup_', backup_timestamp, '.sql');
    
    SET @sql = CONCAT('mysqldump -u root -p education_db > /var/lib/mysql/backups/', backup_file);
    PREPARE stmt FROM @sql;
    EXECUTE stmt;
    DEALLOCATE PREPARE stmt;
    
    INSERT INTO backup_logs (backup_file, backup_time, status)
    VALUES (backup_file, NOW(), 'completed');
END //
DELIMITER ;

-- 创建事件定时备份
CREATE EVENT event_daily_backup
ON SCHEDULE EVERY 1 DAY
STARTS CURRENT_TIMESTAMP
DO
CALL backup_database();

-- 创建恢复存储过程
DELIMITER //
CREATE PROCEDURE restore_database(backup_file VARCHAR(100))
BEGIN
    DECLARE exit_value INT;
    
    SET @sql = CONCAT('mysql -u root -p education_db < /var/lib/mysql/backups/', backup_file);
    SET exit_value = SYSTEM(@sql);
    
    IF exit_value = 0 THEN
        INSERT INTO backup_logs (backup_file, restore_time, status)
        VALUES (backup_file, NOW(), 'success');
    ELSE
        INSERT INTO backup_logs (backup_file, restore_time, status)
        VALUES (backup_file, NOW(), 'failed');
    END IF;
END //
DELIMITER ;
```
